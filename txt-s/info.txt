    // Underline Mouse (Over and Leave) => function
function underlineEffect(text, effect_place) {
    for (let index = 0; index < text.length; index++) {
        text[index].onmouseover = () => {  
            effect_place[index].style.cssText = `
                width: 100%;
                transition: all 1s;
            `;
        };
        text[index].onmouseleave = () => {  
            effect_place[index].style.cssText = `
                width: 0%;
                transition: all 1.5s;
            `;
        };
    }
}
